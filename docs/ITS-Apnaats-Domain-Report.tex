\section{Mastery score: formulas and variables}
%=============================================================================

\label{sec:mastery}
On each \textbf{correct} answer, the server computes an \textbf{instantaneous} mastery estimate
\texttt{newScore} $= M_{\mathrm{inst}}$ using \textbf{session metrics at that moment}:

\begin{mathbox}[Composite mastery $M_{\mathrm{inst}}$ --- \texttt{calculateMastery}]
\begin{equation}
\boxed{
M_{\mathrm{inst}} = 0.40\,A + 0.20\,(1-H) + 0.15\,(1-P_A) + 0.15\,C + 0.10\,T
}
\label{eq:mastery}
\end{equation}
\end{mathbox}

\noindent\textbf{Inputs (symbols match \texttt{masteryService.js}):}

\renewcommand{\arraystretch}{1.4}
\begin{tabularx}{\linewidth}{@{}c l X@{}}
\toprule
\textbf{Sym.} & \textbf{Source in code} & \textbf{Definition} \\
\midrule
$A$ & \texttt{correct / totalAttempts} &
Fraction of \textbf{session} attempts that are correct. If \texttt{totalAttempts}$=0$, $A=0$. \\
\rowcolor{rowhighlight}
$H$ & \texttt{hintsUsed / questionsAnswered} &
Hint rate, capped at $1$. If no questions answered yet, $H=0$. \\
$P_A$ & \texttt{(totalAttempts - questionsAnswered) / totalAttempts} &
``Extra attempts'' beyond one success per question, clipped to $[0,1]$. \\
\rowcolor{rowhighlight}
$C$ & \texttt{learner.confidence\_score} &
Clipped to $[0,1]$. \\
$T$ & \texttt{avgExpectedTime / avgTimeSpent} &
Time efficiency, capped at $1$; if \texttt{avgTimeSpent}$\le 0$, use $T=0.5$. \\
\bottomrule
\end{tabularx}

\noindent Here \texttt{questionsAnswered} $=$ count of completed results in the session;
\texttt{avgTimeSpent} $=$ \texttt{timeSpent / answered}; \texttt{avgExpectedTime} $=$ \texttt{question.expectedTime} for the item just solved.

\paragraph{Worked example (illustrative).}
If \texttt{correct}$=3$, \texttt{totalAttempts}$=5$, \texttt{hintsUsed}$=2$, \texttt{questionsAnswered}$=3$,
$C=0.50$, \texttt{avgTimeSpent}$=50$\,s, \texttt{avgExpectedTime}$=40$\,s: then
$A=0.60$, $H=\min(1,\,2/3)$, $P_A=(5-3)/5=0.40$, $T=\min(1,\,40/50)=0.80$, and
$M_{\mathrm{inst}} \approx 0.552$ (falls in the \textbf{Scholar} band $[0.40,0.70]$).

\paragraph{Level mapping (same thresholds as UI ``Explorer / Scholar / Legend'').}
\[
\texttt{classifyMastery}(M)=
\begin{cases}
\text{Explorer} & M < 0.40 \\
\text{Scholar} & 0.40 \le M \le 0.70 \\
\text{Legend} & M > 0.70
\end{cases}
\qquad
\texttt{difficultyFromMastery}(M)=
\begin{cases}
\texttt{beginner} & M < 0.40 \\
\texttt{intermediate} & M \le 0.70 \\
\texttt{advanced} & M > 0.70
\end{cases}
\]

\paragraph{Stored mastery blend per concept.}
\label{sec:blend}
\textbf{\texttt{updateConceptMastery}} smooths the new estimate into the map:
\begin{equation}
M_{\mathrm{stored}} \leftarrow 0.7\,M_{\mathrm{stored}} + 0.3\,M_{\mathrm{inst}} .
\label{eq:blend}
\end{equation}

\paragraph{Global \texttt{accuracy} (learner object).}
When an answer is correct, the controller updates:
\begin{equation}
\texttt{accuracy} \leftarrow \frac{\texttt{accuracy}\times(\texttt{attempts}-1) + 1}{\texttt{attempts}}
\qquad\text{(clipped to $[0,1]$)}
\label{eq:acclearner}
\end{equation}
\textit{i.e.} a running fraction of ``successful'' learner-level events along the update path (not the same as $A$ unless you align counting).

\section{Pedagogy (summary)}
%=============================================================================

\texttt{applyPedagogicalRule} (\texttt{pedagogyEngine.js}):
\textbf{correct} $\Rightarrow$ praise, \texttt{next\_question}, difficulty hint ``up'' for streak logic;
\textbf{wrong}: 1st \texttt{retry}; 2nd hint L1; 3rd hint L2; $\ge 4$ \texttt{remedial} + difficulty ``down''.
If $t_{\mathrm{spent}}/t_{\mathrm{expected}}>2$ and attempts $\ge 2$, force \textbf{hint level 3}.
